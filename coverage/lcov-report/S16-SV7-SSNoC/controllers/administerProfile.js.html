var directory = require('../models/Directory.js');
var socketsForAdmin = {};//a map of username and socket
 
exports.updateSocket = function (connectedUsers) {
    socketsForAdmin = connectedUsers;
};
 
exports.activate = function(req, res) {
    var name = req.params.userName;
    directory.findName(name, function(user){
        if(!user){
            res.sendStatus(404);
        } else {
            user.updateAccount(true);
            res.sendStatus(200);
            var socket = socketsForAdmin['SSNAdmin'];
            socket.broadcast.emit('send:active', name);
        }
    });
};
 
exports.deactivate = function(req, res) {
    var name = req.params.userName;
    directory.findName(name, function (user) {
        if (!user) {
            res.sendStatus(404);
        } else {
            user.updateAccount(false);
            res.sendStatus(200);
            var socket = socketsForAdmin['SSNAdmin'];
            socket.broadcast.emit('send:inactive', name);
        }
    });
};
 
exports.updateProfile = function(req, res) {
    var name = req.params.userName;
    var data = req.body;
    directory.findName(name, function (user) {
        if (!user) {
            res.sendStatus(404);
        } else {
            if(directory.checkReservedName(data.userName)){
                res.send('reserved');
            } else {
                if (name != data.userName) {
                    directory.findName(data.userName, function (isUsed) {
                        if (!isUsed) {
                            user.updateProfile(data);
                            res.sendStatus(201);
                            var socket = socketsForAdmin['SSNAdmin'];
                            data.oldName = name;
                            socket.broadcast.emit('updateProfile', data);
                        } else {
                            res.send('used');
                        }
                    });
                } else {
                    user.updateProfile(data);
                    res.sendStatus(201);
                    var socket = socketsForAdmin['SSNAdmin'];
                    data.oldName = name;
                    socket.broadcast.emit('updateProfile', data);
                }
            }
        }
    });
};
