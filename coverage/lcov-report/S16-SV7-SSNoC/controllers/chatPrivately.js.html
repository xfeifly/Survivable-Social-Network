function computeRoomId(fromName, toName) {
    var roomIdstring = fromName + toName;
    var roomIdnumber = 0;
    for (var i=0; i< roomIdstring.length; i++){
        roomIdnumber = roomIdnumber + roomIdstring.charCodeAt(i);
    }
    return roomIdnumber;
}
